# --- Desktop UI ---------------------------------------------------------------
PyQt5>=5.15.0
PyQt5-sip>=12.11.0

# --- SFTP / FTPS destination support -----------------------------------------
paramiko>=3.0.0

# --- SMB / CIFS network share support ----------------------------------------
# smbprotocol ships both smbprotocol.* (low-level) and smbclient.* (high-level)
# -- both are used by transport_utils.py and desktop_app.py.
# On Windows the app uses built-in UNC paths as the primary method;
# smbprotocol is the cross-platform fallback and is always installed.
smbprotocol>=1.0.0

# --- WebDAV / Nextcloud / ownCloud destination support -----------------------
# Required for WebDAV destinations (Nextcloud, ownCloud, OneDrive via WebDAV).
# The built-in urllib fallback is insufficient for edge cases.
webdavclient3>=3.14.6

# --- Secure credential storage (recommended) ---------------------------------
# Stores SFTP / FTP / SMB / SMTP passwords in the OS keyring instead of
# plaintext in config.json.  Falls back gracefully if not installed.
#   Windows  : Windows Credential Manager
#   macOS    : macOS Keychain
#   Linux    : SecretService (GNOME Keyring / KWallet)
keyring>=24.0.0

# --- Windows extras (strongly recommended on Windows) ------------------------
# pywin32 enables two important features on Windows:
#   1. Accurate "edited by" file-owner tracking in the change log
#   2. VSS (Volume Shadow Copy) integration for backing up locked/open files
#      (Outlook .pst, browser databases, running app databases, etc.)
# The sys_platform marker means pip skips this automatically on Linux/macOS —
# no need to comment/uncomment anything.
pywin32>=306; sys_platform == "win32"
